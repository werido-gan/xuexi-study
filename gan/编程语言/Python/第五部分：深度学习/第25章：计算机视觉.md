import cv2
import matplotlib.pyplot as plt

# 加载图像
img = cv2.imread('image.jpg')

#### 25.2.3 图像增强

```python
import cv2
import matplotlib.pyplot as plt
import numpy as np

# 加载图像
img = cv2.imread('image.jpg')
img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

# 亮度调整
brightened = cv2.convertScaleAbs(img_rgb, alpha=1.5, beta=0)  # 增加亮度
darkened = cv2.convertScaleAbs(img_rgb, alpha=0.5, beta=0)   # 减少亮度

# 对比度调整
alpha = 1.5  # 对比度增益
beta = 0     # 亮度偏移
adjusted = cv2.convertScaleAbs(img_rgb, alpha=alpha, beta=beta)

# 高斯模糊
blurred = cv2.GaussianBlur(img_rgb, (5, 5), 0)

# 锐化
kernel = np.array([[-1, -1, -1], [-1, 9, -1], [-1, -1, -1]])
sharpened = cv2.filter2D(img_rgb, -1, kernel)

# 显示结果
plt.figure(figsize=(12, 8))
plt.subplot(231), plt.imshow(img_rgb), plt.title('Original')
plt.subplot(232), plt.imshow(brightened), plt.title('Brightened')
plt.subplot(233), plt.imshow(darkened), plt.title('Darkened')
plt.subplot(234), plt.imshow(adjusted), plt.title('Contrast Adjusted')
plt.subplot(235), plt.imshow(blurred), plt.title('Blurred')
plt.subplot(236), plt.imshow(sharpened), plt.title('Sharpened')
plt.tight_layout()
plt.show()
```

### 25.3 图像分类

图像分类是计算机视觉的基础任务，它将图像分类到预定义的类别中。

#### 25.3.1 使用预训练模型

```python
import torch
import torchvision
import torchvision.transforms as transforms
from PIL import Image

# 加载预训练模型
model = torchvision.models.resnet18(pretrained=True)
model.eval()

# 图像预处理
transform = transforms.Compose([
    transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize(
        mean=[0.485, 0.456, 0.406],
        std=[0.229, 0.224, 0.225]
    )
])

# 加载图像
img = Image.open('cat.jpg')
img_transformed = transform(img)
img_batch = img_transformed.unsqueeze(0)  # 添加批次维度

# 预测
with torch.no_grad():
    outputs = model(img_batch)
    _, predicted = torch.max(outputs, 1)

# 加载标签
with open('imagenet_classes.txt') as f:
    classes = [line.strip() for line in f]

print(f'Predicted: {classes[predicted[0]]}')

# 显示图像
import matplotlib.pyplot as plt
plt.imshow(img)
plt.title(f'Predicted: {classes[predicted[0]]}')
plt.axis('off')
plt.show()
```

#### 25.3.2 自定义图像分类模型

```python
import torch
import torch.nn as nn
import torch.optim as optim
import torchvision
import torchvision.transforms as transforms

# 数据准备
transform = transforms.Compose([
    transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize((0.485, 0.456, 0.406), (0.229, 0.224, 0.225))
])

trainset = torchvision.datasets.CIFAR10(root='./data', train=True, download=True, transform=transform)
trainloader = torch.utils.data.DataLoader(trainset, batch_size=32, shuffle=True, num_workers=2)

testset = torchvision.datasets.CIFAR10(root='./data', train=False, download=True, transform=transform)
testloader = torch.utils.data.DataLoader(testset, batch_size=32, shuffle=False, num_workers=2)

classes = ('plane', 'car', 'bird', 'cat', 'deer', 'dog', 'frog', 'horse', 'ship', 'truck')

# 定义模型
class CustomModel(nn.Module):
    def __init__(self):
        super(CustomModel, self).__init__()
        self.conv1 = nn.Conv2d(3, 16, 3, padding=1)
        self.conv2 = nn.Conv2d(16, 32, 3, padding=1)
        self.conv3 = nn.Conv2d(32, 64, 3, padding=1)
        self.pool = nn.MaxPool2d(2, 2)
        self.fc1 = nn.Linear(64 * 28 * 28, 512)
        self.fc2 = nn.Linear(512, 10)
        self.relu = nn.ReLU()
    
    def forward(self, x):
        x = self.pool(self.relu(self.conv1(x)))
        x = self.pool(self.relu(self.conv2(x)))
        x = self.pool(self.relu(self.conv3(x)))
        x = x.view(-1, 64 * 28 * 28)
        x = self.relu(self.fc1(x))
        x = self.fc2(x)
        return x

# 创建模型
model = CustomModel()

# 定义损失函数和优化器
criterion = nn.CrossEntropyLoss()
optimizer = optim.SGD(model.parameters(), lr=0.001, momentum=0.9)

# 训练模型
for epoch in range(10):
    running_loss = 0.0
    for i, data in enumerate(trainloader, 0):
        inputs, labels = data
        
        optimizer.zero_grad()
        
        outputs = model(inputs)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
        
        running_loss += loss.item()
        if i % 100 == 99:
            print(f'[{epoch + 1}, {i + 1:5d}] loss: {running_loss / 100:.3f}')
            running_loss = 0.0

print('Finished Training')

# 测试模型
correct = 0
total = 0
with torch.no_grad():
    for data in testloader:
        images, labels = data
        outputs = model(images)
        _, predicted = torch.max(outputs.data, 1)
        total += labels.size(0)
        correct += (predicted == labels).sum().item()

print(f'Accuracy of the network on the 10000 test images: {100 * correct / total} %')
```

### 25.4 目标检测

目标检测是识别图像中物体的位置和类别的任务。

#### 25.4.1 使用 YOLOv5

```python
import torch
from PIL import Image
import matplotlib.pyplot as plt
import matplotlib.patches as patches

# 加载 YOLOv5 模型
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')

# 加载图像
img = Image.open('street.jpg')

# 预测
results = model(img)

# 显示结果
plt.imshow(img)
ax = plt.gca()

# 绘制边界框
for pred in results.pred[0]:
    x1, y1, x2, y2, conf, cls = pred.tolist()
    if conf > 0.5:  # 置信度阈值
        rect = patches.Rectangle((x1, y1), x2 - x1, y2 - y1, linewidth=2, edgecolor='r', facecolor='none')
        ax.add_patch(rect)
        ax.text(x1, y1, f'{results.names[int(cls)]} {conf:.2f}', bbox=dict(facecolor='white', alpha=0.5))

plt.axis('off')
plt.show()

# 保存结果
results.save()
```

### 25.5 图像分割

图像分割是将图像分割成不同的区域，每个区域对应一个物体或背景。

#### 25.5.1 语义分割

```python
import torch
import torchvision
from PIL import Image
import matplotlib.pyplot as plt
import numpy as np

# 加载模型
model = torchvision.models.segmentation.deeplabv3_resnet101(pretrained=True)
model.eval()

# 图像预处理
transform = torchvision.transforms.Compose([
    torchvision.transforms.ToTensor(),
    torchvision.transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
])

# 加载图像
img = Image.open('street.jpg')
img_tensor = transform(img)

# 预测
with torch.no_grad():
    output = model([img_tensor])['out'][0]

# 获取分割结果
seg_map = output.argmax(0).numpy()

# 颜色映射
palette = torchvision.models.segmentation.deeplabv3_resnet101(pretrained=True).palette

# 应用颜色映射
seg_image = np.zeros((seg_map.shape[0], seg_map.shape[1], 3), dtype=np.uint8)
for label, color in enumerate(palette):
    seg_image[seg_map == label] = color

# 显示结果
plt.figure(figsize=(12, 6))
plt.subplot(121), plt.imshow(img), plt.title('Original')
plt.subplot(122), plt.imshow(seg_image), plt.title('Segmentation')
plt.axis('off')
plt.tight_layout()
plt.show()
```

        # 标签
        coco_labels = torchvision.models.detection.faster_rcnn.COCO_INSTANCE_CATEGORY_NAMES
        ax.text(box[0], box[1], f'{coco_labels[label]} {score:.2f}', bbox=dict(facecolor='white', alpha=0.5))

plt.show()

### 25.7 视频分析

视频分析是处理和理解视频数据的任务。

#### 25.7.1 视频读取和处理

```python
import cv2
import matplotlib.pyplot as plt

# 打开视频
cap = cv2.VideoCapture('video.mp4')

# 检查视频是否成功打开
if not cap.isOpened():
    print("Error opening video file")

# 读取视频信息
fps = cap.get(cv2.CAP_PROP_FPS)
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))

print(f"Video FPS: {fps}")
print(f"Video resolution: {width}x{height}")

# 读取并显示视频帧
while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break
    
    # 转换为 RGB
    frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    
    # 显示帧
    plt.imshow(frame_rgb)
    plt.axis('off')
    plt.pause(1/fps)
    plt.clf()

# 释放资源
cap.release()
plt.close()
```

#### 25.7.2 视频目标检测

```python
import cv2
import torch
from PIL import Image
import numpy as np

# 加载 YOLOv5 模型
model = torch.hub.load('ultralytics/yolov5', 'yolov5s')

# 打开视频
cap = cv2.VideoCapture('video.mp4')

# 检查视频是否成功打开
if not cap.isOpened():
    print("Error opening video file")

# 读取视频信息
fps = cap.get(cv2.CAP_PROP_FPS)
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))

# 创建输出视频
out = cv2.VideoWriter('output.mp4', cv2.VideoWriter_fourcc(*'mp4v'), fps, (width, height))

# 处理视频帧
while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break
    
    # 转换为 PIL 图像
    img = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB))
    
    # 预测
    results = model(img)
    
    # 绘制边界框
    for pred in results.pred[0]:
        x1, y1, x2, y2, conf, cls = pred.tolist()
        if conf > 0.5:
            cv2.rectangle(frame, (int(x1), int(y1)), (int(x2), int(y2)), (0, 255, 0), 2)
            cv2.putText(frame, f'{results.names[int(cls)]} {conf:.2f}', (int(x1), int(y1)-10), cv2.FONT_HERSHEY_SIMPLEX, 0.9, (0, 255, 0), 2)
    
    # 写入输出视频
    out.write(frame)
    
    # 显示帧
    cv2.imshow('Video', frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

# 释放资源
cap.release()
out.release()
cv2.destroyAllWindows()
```
