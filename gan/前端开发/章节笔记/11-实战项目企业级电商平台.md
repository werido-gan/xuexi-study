## 11. 实战项目：企业级电商平台

### 11.1 项目概述

**项目名称**：ShopX — 现代化 B2C 电商平台

**业务功能**：商品浏览、搜索、购物车、订单、支付、用户中心、商家后台

### 11.2 技术选型

```
[技术选型决策树]

前端框架: React 18 + TypeScript
  理由: 团队熟悉度高，生态丰富，TypeScript保证类型安全

状态管理: Zustand
  理由: 轻量，API简单，性能好（避免Redux的样板代码）

路由: React Router v6
  理由: 官方推荐，支持数据路由

UI组件库: Ant Design + Tailwind CSS
  理由: Antd 提供业务组件，Tailwind 处理自定义样式

请求库: Axios + React Query
  理由: React Query 提供缓存、自动重试、乐观更新

构建工具: Vite
  理由: 开发体验极佳，构建速度快

代码规范: ESLint + Prettier + Husky
  理由: 自动化规范，减少Code Review中的格式问题

测试: Vitest + Testing Library
  理由: Vite生态的测试方案，API与Jest兼容

部署: Docker + Nginx + GitHub Actions CI/CD
  理由: 标准化部署，自动化流程
```

### 11.5 核心代码示例

#### 商品列表页（含虚拟滚动）

```typescript
// src/pages/ProductList/index.tsx
import { useState, useCallback } from 'react';
import { useInfiniteQuery } from 'react-query';
import { useVirtualizer } from '@tanstack/react-virtual';
import { ProductCard } from '@/components/business/ProductCard';
import { useDebounce } from '@/hooks/useDebounce';
import { fetchProducts } from '@/api/product';

interface FilterState {
  keyword: string;
  categoryId: string;
  sortBy: 'price_asc' | 'price_desc' | 'sales' | 'newest';
  priceRange: [number, number];
}

export default function ProductListPage() {
  const [filter, setFilter] = useState<FilterState>({
    keyword: '',
    categoryId: '',
    sortBy: 'newest',
    priceRange: [0, 99999]
  });
  
  // 防抖处理搜索输入
  const debouncedKeyword = useDebounce(filter.keyword, 500);
  
  // 无限滚动 + 请求缓存
  const {
    data,
    fetchNextPage,
    hasNextPage,
    isFetchingNextPage,
    isLoading
  } = useInfiniteQuery({
    queryKey: ['products', { ...filter, keyword: debouncedKeyword }],
    queryFn: ({ pageParam = 1 }) => fetchProducts({
      page: pageParam,
      ...filter,
      keyword: debouncedKeyword
    }),
    getNextPageParam: (lastPage, pages) =>
      lastPage.hasMore ? pages.length + 1 : undefined,
    staleTime: 5 * 60 * 1000  // 5分钟内不重新请求
  });
  
  const products = data?.pages.flatMap(p => p.items) ?? [];
  
  // 虚拟滚动（只渲染可视区域内的DOM）
  const parentRef = useRef<HTMLDivElement>(null);
  const virtualizer = useVirtualizer({
    count: products.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 300,  // 每项预估高度
  });
  
  return (
    <div className="product-list-page">
      <SearchBar 
        value={filter.keyword}
        onChange={keyword => setFilter(prev => ({ ...prev, keyword }))}
      />
      
      <div ref={parentRef} style={{ height: '800px', overflow: 'auto' }}>
        <div style={{ height: virtualizer.getTotalSize() }}>
          {virtualizer.getVirtualItems().map(virtualItem => (
            <div
              key={virtualItem.key}
              style={{
                position: 'absolute',
                top: virtualItem.start,
                width: '100%'
              }}
            >
              <ProductCard product={products[virtualItem.index]} />
            </div>
          ))}
        </div>
      </div>
      
      {hasNextPage && (
        <button onClick={() => fetchNextPage()} disabled={isFetchingNextPage}>
          {isFetchingNextPage ? '加载中...' : '加载更多'}
        </button>
      )}
    </div>
  );
}
```

#### 购物车状态管理

```typescript
// src/store/useCartStore.ts
import { create } from 'zustand';
import { persist } from 'zustand/middleware';  // 持久化到localStorage

interface CartItem {
  id: string;
  name: string;
  price: number;
  quantity: number;
  imageUrl: string;
  skuId: string;
}

interface CartStore {
  items: CartItem[];
  
  // 派生数据（computed）
  totalCount: () => number;
  totalPrice: () => number;
  
  // 操作方法
  addItem: (item: Omit<CartItem, 'quantity'>) => void;
  removeItem: (skuId: string) => void;
  updateQuantity: (skuId: string, quantity: number) => void;
  clearCart: () => void;
}

const useCartStore = create<CartStore>()(
  persist(
    (set, get) => ({
      items: [],
      
      totalCount: () => get().items.reduce((sum, item) => sum + item.quantity, 0),
      totalPrice: () => get().items.reduce((sum, item) => sum + item.price * item.quantity, 0),
      
      addItem: (newItem) => set(state => {
        const existing = state.items.find(item => item.skuId === newItem.skuId);
        if (existing) {
          return {
            items: state.items.map(item =>
              item.skuId === newItem.skuId
                ? { ...item, quantity: item.quantity + 1 }
                : item
            )
          };
        }
        return { items: [...state.items, { ...newItem, quantity: 1 }] };
      }),
      
      removeItem: (skuId) => set(state => ({
        items: state.items.filter(item => item.skuId !== skuId)
      })),
      
      updateQuantity: (skuId, quantity) => set(state => ({
        items: quantity <= 0
          ? state.items.filter(item => item.skuId !== skuId)
          : state.items.map(item =>
              item.skuId === skuId ? { ...item, quantity } : item
            )
      })),
      
      clearCart: () => set({ items: [] })
    }),
    {
      name: 'cart-storage',              // localStorage key
      partialize: state => ({ items: state.items })  // 只持久化items
    }
  )
);

export default useCartStore;
```

#### CI/CD 配置

```yaml
# .github/workflows/ci.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  # 代码质量检查
  quality:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: '18'
          cache: 'pnpm'
      
      - run: pnpm install --frozen-lockfile
      - run: pnpm lint          # ESLint检查
      - run: pnpm type-check    # TypeScript类型检查
      - run: pnpm test          # 单元测试

  # 构建
  build:
    runs-on: ubuntu-latest
    needs: quality
    steps:
      - uses: actions/checkout@v3
      - run: pnpm install --frozen-lockfile
      - run: pnpm build
      - uses: actions/upload-artifact@v3
        with:
          name: dist
          path: dist/

  # 部署到生产（只在main分支）
  deploy:
    runs-on: ubuntu-latest
    needs: build
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/download-artifact@v3
        with:
          name: dist
      
      - name: Deploy to Server
        uses: appleboy/ssh-action@master
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USER }}
          key: ${{ secrets.SSH_PRIVATE_KEY }}
          script: |
            cd /var/www/shopx
            rsync -avz dist/ .
            nginx -s reload
```
