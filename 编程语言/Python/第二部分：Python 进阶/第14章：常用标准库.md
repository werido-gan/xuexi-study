### 14.1 datetime - 日期时间处理

```python
from datetime import datetime, date, time, timedelta

# 获取当前时间
now = datetime.now()
print(now)  # 2024-01-01 12:00:00

# 创建指定时间
d = date(2024, 1, 1)
t = time(12, 0, 0)
dt = datetime(2024, 1, 1, 12, 0, 0)

# 格式化
print(now.strftime("%Y-%m-%d %H:%M:%S"))  # 2024-01-01 12:00:00

# 解析字符串
str_time = "2024-01-01 12:00:00"
dt = datetime.strptime(str_time, "%Y-%m-%d %H:%M:%S")

# 时间差
delta = timedelta(days=1, hours=2)
new_time = now + delta
print(new_time)
```

### 14.2 collections - 高级数据结构

```python
from collections import Counter, defaultdict, OrderedDict, deque

# Counter - 计数
fruits = ["apple", "banana", "apple", "orange", "banana", "apple"]
counter = Counter(fruits)
print(counter)  # Counter({'apple': 3, 'banana': 2, 'orange': 1})
print(counter.most_common(2))  # [('apple', 3), ('banana', 2)]

# defaultdict - 默认值字典
d = defaultdict(int)
d["apple"] += 1
d["banana"] += 1
print(d)  # defaultdict(<class 'int'>, {'apple': 1, 'banana': 1})

# deque - 双端队列
q = deque()
q.append(1)
q.append(2)
q.appendleft(0)
print(q)  # deque([0, 1, 2])
q.popleft()  # 0
print(q)  # deque([1, 2])
```

### 14.3 random - 随机数

```python
import random

# 生成随机数
print(random.random())  # 0-1之间的随机浮点数
print(random.randint(1, 10))  # 1-10之间的随机整数
print(random.choice([1, 2, 3, 4, 5]))  # 从列表中随机选择一个

# 打乱列表
lst = [1, 2, 3, 4, 5]
random.shuffle(lst)
print(lst)

# 生成随机样本
print(random.sample(range(100), 5))  # 从0-99中随机选5个不重复的数
```

### 14.4 os - 操作系统接口

```python
import os
import os.path

# 环境变量
print(os.environ.get("PATH"))

# 工作目录
print(os.getcwd())
os.chdir("/path/to/dir")

# 文件操作
print(os.path.exists("file.txt"))
print(os.path.isfile("file.txt"))
print(os.path.isdir("dir"))
print(os.path.join("a", "b", "c"))
print(os.path.splitext("file.txt"))  # ('file', '.txt')

# 目录操作
os.makedirs("dir/subdir", exist_ok=True)
os.rmdir("dir/subdir")  # 只能删除空目录

# 遍历目录
for root, dirs, files in os.walk("."):
    for file in files:
        print(os.path.join(root, file))
```

### 14.6 json - JSON 处理

```python
import json

# Python 对象转 JSON
obj = {"name": "小明", "age": 18, "hobbies": ["篮球", "编程"]}
json_str = json.dumps(obj, ensure_ascii=False, indent=2)
print(json_str)

# JSON 转 Python 对象
json_str = '{"name": "小明", "age": 18}'
obj = json.loads(json_str)
print(obj["name"])

# 读写文件
with open("data.json", "w", encoding="utf-8") as f:
    json.dump(obj, f, ensure_ascii=False, indent=2)

with open("data.json", "r", encoding="utf-8") as f:
    obj = json.load(f)
```

### 14.7 csv - CSV 文件处理

```python
import csv

# 写入 CSV
with open("data.csv", "w", newline="", encoding="utf-8") as f:
    writer = csv.writer(f)
    writer.writerow(["姓名", "年龄", "城市"])
    writer.writerow(["小明", 18, "北京"])
    writer.writerow(["小红", 20, "上海"])

# 读取 CSV
with open("data.csv", "r", encoding="utf-8") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)

# 使用 DictReader
with open("data.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["姓名"], row["年龄"])
```

### 14.9 itertools - 迭代工具

```python
import itertools

# 无限迭代器
for i in itertools.count(10, 2):  # 从10开始，步长2
    if i > 20:
        break
    print(i)  # 10, 12, 14, 16, 18, 20

# 循环迭代器
for i in itertools.cycle(["A", "B", "C"]):
    if i == "C":
        break
    print(i)  # A, B

# 重复迭代器
for i in itertools.repeat("Hello", 3):
    print(i)  # Hello, Hello, Hello

# 组合
print(list(itertools.permutations([1, 2, 3], 2)))  # 排列
print(list(itertools.combinations([1, 2, 3], 2)))  # 组合

# 连接
print(list(itertools.chain([1, 2], [3, 4])))  # [1, 2, 3, 4]
```

### 14.10 functools - 函数工具

```python
import functools

# 偏函数
int2 = functools.partial(int, base=2)  # 固定base参数为2
print(int2("1010"))  # 10

# 装饰器工具
@functools.lru_cache(maxsize=128)
def fib(n):
    if n <= 1:
        return n
    return fib(n-1) + fib(n-2)

print(fib(10))  # 55，使用缓存提高性能

# 比较函数
lst = [("a", 3), ("b", 1), ("c", 2)]
lst.sort(key=functools.cmp_to_key(lambda x, y: x[1] - y[1]))
print(lst)  # [('b', 1), ('c', 2), ('a', 3)]
```
