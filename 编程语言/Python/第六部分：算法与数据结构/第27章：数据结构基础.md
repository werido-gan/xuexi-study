### 27.1 数据结构简介

数据结构是组织和存储数据的方式，它决定了如何访问和修改数据。

> **数据结构的重要性**：

### 27.2 线性数据结构

#### 27.2.1 数组（Array）

数组是一种连续的内存空间，用于存储相同类型的数据。

```python
# 数组的基本操作
arr = [1, 2, 3, 4, 5]

# 访问元素
print(f"arr[0]: {arr[0]}")  # 1

# 修改元素
arr[0] = 10
print(f"arr[0]: {arr[0]}")  # 10

# 添加元素
arr.append(6)
print(f"arr: {arr}")  # [10, 2, 3, 4, 5, 6]

# 插入元素
arr.insert(1, 20)
print(f"arr: {arr}")  # [10, 20, 2, 3, 4, 5, 6]

# 删除元素
arr.remove(20)
print(f"arr: {arr}")  # [10, 2, 3, 4, 5, 6]

# 弹出元素
popped = arr.pop()
print(f"Popped: {popped}, arr: {arr}")  # Popped: 6, arr: [10, 2, 3, 4, 5]

# 长度
print(f"Length: {len(arr)}")  # 5

# 切片
print(f"arr[1:4]: {arr[1:4]}")  # [2, 3, 4]

# 遍历
for item in arr:
    print(item, end=" ")  # 10 2 3 4 5
print()
```

#### 27.2.2 链表（Linked List）

链表是一种非连续的内存空间，通过指针连接各个节点。

```python
# 链表的实现
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class LinkedList:
    def __init__(self):
        self.head = None
    
    def append(self, val):
        """添加元素到链表末尾"""
        if not self.head:
            self.head = ListNode(val)
            return
        curr = self.head
        while curr.next:
            curr = curr.next
        curr.next = ListNode(val)
    
    def prepend(self, val):
        """添加元素到链表头部"""
        new_node = ListNode(val)
        new_node.next = self.head
        self.head = new_node
    
    def delete(self, val):
        """删除指定值的节点"""
        if not self.head:
            return
        if self.head.val == val:
            self.head = self.head.next
            return
        curr = self.head
        while curr.next:
            if curr.next.val == val:
                curr.next = curr.next.next
                return
            curr = curr.next
    
    def search(self, val):
        """查找指定值的节点"""
        curr = self.head
        while curr:
            if curr.val == val:
                return True
            curr = curr.next
        return False
    
    def display(self):
        """显示链表"""
        curr = self.head
        while curr:
            print(curr.val, end=" -> ")
            curr = curr.next
        print("None")

# 测试链表
ll = LinkedList()
ll.append(1)
ll.append(2)
ll.append(3)
ll.display()  # 1 -> 2 -> 3 -> None

ll.prepend(0)
ll.display()  # 0 -> 1 -> 2 -> 3 -> None

ll.delete(2)
ll.display()  # 0 -> 1 -> 3 -> None

print(f"Search 3: {ll.search(3)}")  # True
print(f"Search 2: {ll.search(2)}")  # False
```

#### 27.2.3 栈（Stack）

栈是一种后进先出（LIFO）的数据结构。

```python
# 使用列表实现栈
class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        """入栈"""
        self.items.append(item)
    
    def pop(self):
        """出栈"""
        if not self.is_empty():
            return self.items.pop()
        return None
    
    def peek(self):
        """查看栈顶元素"""
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        """判断栈是否为空"""
        return len(self.items) == 0
    
    def size(self):
        """获取栈的大小"""
        return len(self.items)

# 测试栈
s = Stack()
s.push(1)
s.push(2)
s.push(3)
print(f"Stack size: {s.size()}")  # 3
print(f"Top element: {s.peek()}")  # 3

print(f"Pop: {s.pop()}")  # 3
print(f"Pop: {s.pop()}")  # 2
print(f"Stack size: {s.size()}")  # 1
print(f"Is empty: {s.is_empty()}")  # False

print(f"Pop: {s.pop()}")  # 1
print(f"Is empty: {s.is_empty()}")  # True
```

#### 27.2.4 队列（Queue）

队列是一种先进先出（FIFO）的数据结构。

```python
# 使用列表实现队列
class Queue:
    def __init__(self):
        self.items = []
    
    def enqueue(self, item):
        """入队"""
        self.items.append(item)
    
    def dequeue(self):
        """出队"""
        if not self.is_empty():
            return self.items.pop(0)
        return None
    
    def front(self):
        """查看队首元素"""
        if not self.is_empty():
            return self.items[0]
        return None
    
    def is_empty(self):
        """判断队列是否为空"""
        return len(self.items) == 0
    
    def size(self):
        """获取队列的大小"""
        return len(self.items)

# 测试队列
q = Queue()
q.enqueue(1)
q.enqueue(2)
q.enqueue(3)
print(f"Queue size: {q.size()}")  # 3
print(f"Front element: {q.front()}")  # 1

print(f"Dequeue: {q.dequeue()}")  # 1
print(f"Dequeue: {q.dequeue()}")  # 2
print(f"Queue size: {q.size()}")  # 1
print(f"Is empty: {q.is_empty()}")  # False

print(f"Dequeue: {q.dequeue()}")  # 3
print(f"Is empty: {q.is_empty()}")  # True

# 使用 collections.deque 实现队列
from collections import deque

q = deque()
q.append(1)  # 入队
q.append(2)
q.append(3)
print(f"Queue: {list(q)}")  # [1, 2, 3]

print(f"Dequeue: {q.popleft()}")  # 1
print(f"Queue: {list(q)}")  # [2, 3]
print(f"Front element: {q[0]}")  # 2
print(f"Queue size: {len(q)}")  # 2
print(f"Is empty: {len(q) == 0}")  # False
```

### 27.3 非线性数据结构

#### 27.3.1 树（Tree）

树是一种层次结构，由节点组成，每个节点可以有多个子节点。

```python
# 二叉树的实现
class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

class BinaryTree:
    def __init__(self):
        self.root = None
    
    def insert(self, val):
        """插入节点"""
        if not self.root:
            self.root = TreeNode(val)
            return
        
        def _insert(node, val):
            if val < node.val:
                if not node.left:
                    node.left = TreeNode(val)
                else:
                    _insert(node.left, val)
            else:
                if not node.right:
                    node.right = TreeNode(val)
                else:
                    _insert(node.right, val)
        
        _insert(self.root, val)
    
    def inorder_traversal(self):
        """中序遍历"""
        result = []
        
        def _inorder(node):
            if node:
                _inorder(node.left)
                result.append(node.val)
                _inorder(node.right)
        
        _inorder(self.root)
        return result
    
    def preorder_traversal(self):
        """前序遍历"""
        result = []
        
        def _preorder(node):
            if node:
                result.append(node.val)
                _preorder(node.left)
                _preorder(node.right)
        
        _preorder(self.root)
        return result
    
    def postorder_traversal(self):
        """后序遍历"""
        result = []
        
        def _postorder(node):
            if node:
                _postorder(node.left)
                _postorder(node.right)
                result.append(node.val)
        
        _postorder(self.root)
        return result

# 测试二叉树
tree = BinaryTree()
tree.insert(5)
tree.insert(3)
tree.insert(7)
tree.insert(2)
tree.insert(4)
tree.insert(6)
tree.insert(8)

print(f"Inorder traversal: {tree.inorder_traversal()}")  # [2, 3, 4, 5, 6, 7, 8]
print(f"Preorder traversal: {tree.preorder_traversal()}")  # [5, 3, 2, 4, 7, 6, 8]
print(f"Postorder traversal: {tree.postorder_traversal()}")  # [2, 4, 3, 6, 8, 7, 5]
```

#### 27.3.2 二叉搜索树（Binary Search Tree）

二叉搜索树是一种特殊的二叉树，左子树的所有节点值小于根节点，右子树的所有节点值大于根节点。

```python
# 二叉搜索树的实现
class BST:
    def __init__(self):
        self.root = None
    
    def insert(self, val):
        """插入节点"""
        if not self.root:
            self.root = TreeNode(val)
            return
        
        def _insert(node, val):
            if val < node.val:
                if not node.left:
                    node.left = TreeNode(val)
                else:
                    _insert(node.left, val)
            elif val > node.val:
                if not node.right:
                    node.right = TreeNode(val)
                else:
                    _insert(node.right, val)
            # 忽略重复值
        
        _insert(self.root, val)
    
    def search(self, val):
        """搜索节点"""
        def _search(node, val):
            if not node:
                return False
            if node.val == val:
                return True
            elif val < node.val:
                return _search(node.left, val)
            else:
                return _search(node.right, val)
        
        return _search(self.root, val)
    
    def delete(self, val):
        """删除节点"""
        def _find_min(node):
            while node.left:
                node = node.left
            return node
        
        def _delete(node, val):
            if not node:
                return node
            if val < node.val:
                node.left = _delete(node.left, val)
            elif val > node.val:
                node.right = _delete(node.right, val)
            else:
                # 叶子节点
                if not node.left and not node.right:
                    return None
                # 只有一个子节点
                elif not node.left:
                    return node.right
                elif not node.right:
                    return node.left
                # 有两个子节点
                else:
                    min_node = _find_min(node.right)
                    node.val = min_node.val
                    node.right = _delete(node.right, min_node.val)
            return node
        
        self.root = _delete(self.root, val)

# 测试二叉搜索树
bst = BST()
bst.insert(5)
bst.insert(3)
bst.insert(7)
bst.insert(2)
bst.insert(4)
bst.insert(6)
bst.insert(8)

print(f"Search 4: {bst.search(4)}")  # True
print(f"Search 9: {bst.search(9)}")  # False

print(f"Inorder traversal before deletion: {bst.inorder_traversal()}")  # [2, 3, 4, 5, 6, 7, 8]

bst.delete(5)
print(f"Inorder traversal after deletion: {bst.inorder_traversal()}")  # [2, 3, 4, 6, 7, 8]
```

#### 27.3.3 堆（Heap）

堆是一种特殊的完全二叉树，分为最大堆和最小堆。

```python
# 使用 heapq 模块实现堆
import heapq

# 最小堆
min_heap = []
heapq.heappush(min_heap, 3)
heapq.heappush(min_heap, 1)
heapq.heappush(min_heap, 4)
heapq.heappush(min_heap, 1)
heapq.heappush(min_heap, 5)

print(f"Min heap: {min_heap}")  # [1, 1, 4, 3, 5]
print(f"Pop: {heapq.heappop(min_heap)}")  # 1
print(f"Pop: {heapq.heappop(min_heap)}")  # 1
print(f"Min heap after pops: {min_heap}")  # [3, 5, 4]

# 最大堆（使用负数）
max_heap = []
heapq.heappush(max_heap, -3)
heapq.heappush(max_heap, -1)
heapq.heappush(max_heap, -4)
heapq.heappush(max_heap, -1)
heapq.heappush(max_heap, -5)

print(f"Max heap (stored as negatives): {max_heap}")  # [-5, -3, -4, -1, -1]
print(f"Pop: {-heapq.heappop(max_heap)}")  # 5
print(f"Pop: {-heapq.heappop(max_heap)}")  # 4
print(f"Max heap after pops: {[-x for x in max_heap]}")  # [3, 1, 1]

# 堆排序
def heap_sort(arr):
    heap = []
    for item in arr:
        heapq.heappush(heap, item)
    return [heapq.heappop(heap) for _ in range(len(heap))]

arr = [3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5]
sorted_arr = heap_sort(arr)
print(f"Original array: {arr}")
print(f"Sorted array: {sorted_arr}")
```

#### 27.3.4 图（Graph）

图是由节点和边组成的数据结构。

```python
# 邻接矩阵表示图
class GraphMatrix:
    def __init__(self, num_vertices):
        self.num_vertices = num_vertices
        self.matrix = [[0] * num_vertices for _ in range(num_vertices)]
    
    def add_edge(self, u, v, weight=1):
        """添加边"""
        self.matrix[u][v] = weight
        # 如果是无向图，添加反向边
        # self.matrix[v][u] = weight
    
    def remove_edge(self, u, v):
        """删除边"""
        self.matrix[u][v] = 0
        # 如果是无向图，删除反向边
        # self.matrix[v][u] = 0
    
    def print_graph(self):
        """打印图"""
        for i in range(self.num_vertices):
            edges = []
            for j in range(self.num_vertices):
                if self.matrix[i][j] != 0:
                    edges.append(f"{j}({self.matrix[i][j]})")
            print(f"{i}: {' '.join(edges)}")

# 测试邻接矩阵图
g = GraphMatrix(5)
g.add_edge(0, 1, 2)
g.add_edge(0, 3, 5)
g.add_edge(1, 2, 3)
g.add_edge(2, 3, 1)
g.add_edge(3, 4, 4)
g.print_graph()

# 邻接表表示图
class GraphAdjList:
    def __init__(self):
        self.adj_list = {}
    
    def add_vertex(self, vertex):
        """添加顶点"""
        if vertex not in self.adj_list:
            self.adj_list[vertex] = []
    
    def add_edge(self, u, v, weight=1):
        """添加边"""
        self.add_vertex(u)
        self.add_vertex(v)
        self.adj_list[u].append((v, weight))
        # 如果是无向图，添加反向边
        # self.adj_list[v].append((u, weight))
    
    def remove_edge(self, u, v):
        """删除边"""
        if u in self.adj_list:
            self.adj_list[u] = [(vertex, w) for vertex, w in self.adj_list[u] if vertex != v]
        # 如果是无向图，删除反向边
        # if v in self.adj_list:
        #     self.adj_list[v] = [(vertex, w) for vertex, w in self.adj_list[v] if vertex != u]
    
    def remove_vertex(self, vertex):
        """删除顶点"""
        if vertex in self.adj_list:
            del self.adj_list[vertex]
            for v in self.adj_list:
                self.adj_list[v] = [(u, w) for u, w in self.adj_list[v] if u != vertex]
    
    def print_graph(self):
        """打印图"""
        for vertex, edges in self.adj_list.items():
            edge_str = ' '.join([f"{v}({w})" for v, w in edges])
            print(f"{vertex}: {edge_str}")

# 测试邻接表图
g = GraphAdjList()
g.add_edge('A', 'B', 2)
g.add_edge('A', 'D', 5)
g.add_edge('B', 'C', 3)
g.add_edge('C', 'D', 1)
g.add_edge('D', 'E', 4)
g.print_graph()

# 图的遍历：深度优先搜索（DFS）
def dfs(graph, start, visited=None):
    if visited is None:
        visited = set()
    visited.add(start)
    print(start, end=" ")
    for neighbor, _ in graph.adj_list[start]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)

print("\nDFS traversal:")
dfs(g, 'A')
print()

# 图的遍历：广度优先搜索（BFS）
def bfs(graph, start):
    visited = set()
    queue = [start]
    visited.add(start)
    
    while queue:
        vertex = queue.pop(0)
        print(vertex, end=" ")
        for neighbor, _ in graph.adj_list[vertex]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)

print("BFS traversal:")
bfs(g, 'A')
print()
```

### 27.4 哈希表（Hash Table）

哈希表是一种通过哈希函数将键映射到值的数据结构。

```python
# 使用字典实现哈希表
# Python 的字典就是哈希表的实现

# 基本操作
hash_table = {}

# 添加键值对
hash_table['name'] = 'Alice'
hash_table['age'] = 25
hash_table['city'] = 'New York'
print(f"Hash table: {hash_table}")

# 访问值
print(f"Name: {hash_table['name']}")  # Alice
print(f"Age: {hash_table.get('age')}")  # 25
print(f"Gender: {hash_table.get('gender', 'Unknown')}")  # Unknown

# 修改值
hash_table['age'] = 26
print(f"Updated age: {hash_table['age']}")  # 26

# 删除键值对
del hash_table['city']
print(f"Hash table after deletion: {hash_table}")

# 检查键是否存在
print(f"'name' in hash_table: {'name' in hash_table}")  # True
print(f"'city' in hash_table: {'city' in hash_table}")  # False

# 获取所有键
print(f"Keys: {list(hash_table.keys())}")  # ['name', 'age']

# 获取所有值
print(f"Values: {list(hash_table.values())}")  # ['Alice', 26]

# 获取所有键值对
print(f"Items: {list(hash_table.items())}")  # [('name', 'Alice'), ('age', 26)]

# 哈希表的应用：计数
from collections import defaultdict

def count_frequency(arr):
    freq = defaultdict(int)
    for item in arr:
        freq[item] += 1
    return freq

arr = [1, 2, 3, 2, 1, 3, 1, 4, 2]
freq = count_frequency(arr)
print(f"Frequency: {dict(freq)}")  # {1: 3, 2: 3, 3: 2, 4: 1}

# 哈希表的应用：两数之和
def two_sum(nums, target):
    hash_map = {}
    for i, num in enumerate(nums):
        complement = target - num
        if complement in hash_map:
            return [hash_map[complement], i]
        hash_map[num] = i
    return []

nums = [2, 7, 11, 15]
target = 9
print(f"Two sum indices: {two_sum(nums, target)}")  # [0, 1]
```

### 27.5 数据结构的选择

| 数据结构 | 优点 | 缺点 | 适用场景 |
|---------|------|------|----------|
| 数组 | 随机访问快，内存连续 | 插入删除慢，大小固定 | 需要频繁随机访问的场景 |
| 链表 | 插入删除快，大小灵活 | 随机访问慢，内存不连续 | 需要频繁插入删除的场景 |
| 栈 | 后进先出，操作简单 | 只能访问栈顶元素 | 函数调用栈、表达式求值 |
| 队列 | 先进先出，操作简单 | 只能访问队首元素 | 任务队列、消息队列 |
| 树 | 层次结构，查找效率高 | 实现复杂 | 二叉搜索树、堆、字典树 |
| 图 | 表达复杂关系 | 实现复杂 | 社交网络、路径规划 |
| 哈希表 | 查找、插入、删除快 | 哈希冲突，空间消耗大 | 缓存、计数器、映射 |

### 27.6 时间复杂度和空间复杂度

#### 27.6.1 时间复杂度

时间复杂度是衡量算法执行时间随输入规模增长的增长率。

常见的时间复杂度：
- O(1)：常数时间
- O(log n)：对数时间
- O(n)：线性时间
- O(n log n)：线性对数时间
- O(n²)：平方时间
- O(2ⁿ)：指数时间
- O(n!)：阶乘时间

#### 27.6.2 空间复杂度

空间复杂度是衡量算法所需存储空间随输入规模增长的增长率。

常见的空间复杂度：
- O(1)：常数空间
- O(log n)：对数空间
- O(n)：线性空间
- O(n²)：平方空间

#### 27.6.3 常见数据结构的时间复杂度

| 数据结构 | 访问 | 搜索 | 插入 | 删除 |
|---------|------|------|------|------|
| 数组 | O(1) | O(n) | O(n) | O(n) |
| 链表 | O(n) | O(n) | O(1) | O(1) |
| 栈 | O(n) | O(n) | O(1) | O(1) |
| 队列 | O(n) | O(n) | O(1) | O(1) |
| 二叉搜索树 | O(log n) | O(log n) | O(log n) | O(log n) |
| 堆 | O(1) | O(n) | O(log n) | O(log n) |
| 哈希表 | O(1) | O(1) | O(1) | O(1) |

### 27.7 数据结构的应用

1. **数组**：用于存储和访问连续的数据，如数字列表、字符串等
2. **链表**：用于实现栈、队列、哈希表等数据结构
3. **栈**：用于函数调用、表达式求值、括号匹配等
4. **队列**：用于任务调度、广度优先搜索等
5. **树**：用于实现二叉搜索树、堆、字典树等
